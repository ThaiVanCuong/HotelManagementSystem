225 // Get Real Path when upload photo(from uri - image/mame_image)
226 public String getRealPathFromURI(Uri contentUri) {
227 String path = null;
228 String[] proj = {MediaStore.MediaColumns.DATA};
229 Cursor cursor = getContentResolver().query(contentUri, proj, null, null, null);
230 if (cursor.moveToFirst()) {
231 int column_index = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DATA);
232 path = cursor.getString(column_index);
233 }
234 cursor.close();
235 return path;
236 }
