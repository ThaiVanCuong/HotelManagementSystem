172 public static boolean isEmailValid(EditText editText) {
173 String email = editText.getText().toString();
174 if (email.equals("")) return true;
175 String expression = "^[\\w\\.-]+@([\\w\\-]+\\.)+[A-Z]+$";
176 Pattern pattern = Pattern.compile(expression, Pattern.CASE_INSENSITIVE);
177 Matcher matcher = pattern.matcher(email);
178 return matcher.matches();
179 }

326 super.onActivityResult(requestCode, resultCode, data);
327 if (resultCode == RESULT_OK) {
328 if (requestCode == REQUEST_CHOOSE_PHOTO) {

389 // Get Real Path when upload photo(from uri - image/mame_image)
390 public String getRealPathFromURI(Uri contentUri) {
391 String path = null;
392 String[] proj = {MediaStore.MediaColumns.DATA};
393 Cursor cursor = getContentResolver().query(contentUri, proj, null, null, null);
394 if (cursor.moveToFirst()) {
395 int column_index = cursor.getColumnIndexOrThrow(MediaStore.MediaColumns.DATA);
396 path = cursor.getString(column_index);
397 }
398 cursor.close();
399 return path;
400 }
